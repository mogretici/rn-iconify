# Components

API reference for all icon components.

## Icon Components

rn-iconify exports 200+ icon set components. Each component renders icons from its specific icon set.

### Common Props

All icon components accept these props:

```tsx
interface IconProps {
  /** Icon name (required) */
  name: string;

  /** Icon size in pixels */
  size?: number;

  /** Icon color */
  color?: string;

  /** Rotation angle */
  rotate?: 0 | 90 | 180 | 270;

  /** Flip direction */
  flip?: 'horizontal' | 'vertical' | 'both';

  /** Animation preset */
  animation?: 'spin' | 'pulse' | 'bounce' | 'shake' | 'ping' | 'wiggle';

  /** Placeholder while loading */
  placeholder?: ReactNode | 'skeleton' | 'pulse' | 'shimmer';

  /** Additional styles */
  style?: ViewStyle;

  /** Accessibility label */
  accessibilityLabel?: string;
}
```

### Default Values

| Prop | Default |
|------|---------|
| `size` | `24` |
| `color` | `'#000000'` |
| `rotate` | `0` |
| `flip` | `undefined` |
| `animation` | `undefined` |
| `placeholder` | `undefined` |

## IconThemeProvider

Provides global defaults for all icon components.

```tsx copy
import { IconThemeProvider } from 'rn-iconify';

<IconThemeProvider theme={theme}>
  {children}
</IconThemeProvider>
```

### Props

```tsx
interface IconThemeProviderProps {
  /** Theme configuration */
  theme: IconTheme;

  /** Child components */
  children: ReactNode;
}

interface IconTheme {
  /** Default size */
  size?: number;

  /** Default color */
  color?: string;

  /** Default placeholder */
  placeholder?: ReactNode | 'skeleton' | 'pulse' | 'shimmer';
}
```

### Example

```tsx copy
<IconThemeProvider
  theme={{
    size: 24,
    color: '#333',
    placeholder: 'shimmer',
  }}
>
  <App />
</IconThemeProvider>
```

## Generic Icon Component

For dynamic icon rendering across sets:

```tsx copy
import { Icon } from 'rn-iconify';

<Icon name="mdi:home" size={24} />
<Icon name="heroicons:user" size={24} />
```

### Props

```tsx
interface GenericIconProps extends IconProps {
  /** Full icon name with prefix (e.g., "mdi:home") */
  name: string;
}
```

### Example

```tsx copy
const iconName = 'mdi:home'; // Dynamic

<Icon name={iconName} size={24} color="blue" />
```
