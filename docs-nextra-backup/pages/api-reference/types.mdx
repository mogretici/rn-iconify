# Types

TypeScript type definitions exported by rn-iconify.

## Icon Name Types

Each icon set exports its own icon name type for full autocomplete support.

### MdiIconName

```tsx copy
import { type MdiIconName } from 'rn-iconify';

const icon: MdiIconName = 'home'; // ✅
const invalid: MdiIconName = 'invalid'; // ❌ TypeScript error
```

## Component Props Types

### IconProps

Base props shared by all icon components.

```tsx copy
import { type IconProps } from 'rn-iconify';

interface IconProps {
  /** Icon name */
  name: string;

  /** Icon size in pixels */
  size?: number;

  /** Icon color (any valid color string) */
  color?: string;

  /** Rotation angle in degrees */
  rotate?: 0 | 90 | 180 | 270;

  /** Flip direction */
  flip?: 'horizontal' | 'vertical' | 'both';

  /** Animation preset */
  animation?: AnimationType;

  /** Loading placeholder */
  placeholder?: PlaceholderType;

  /** Additional styles */
  style?: ViewStyle;

  /** Accessibility label */
  accessibilityLabel?: string;

  /** Accessible (for screen readers) */
  accessible?: boolean;

  /** Accessibility role */
  accessibilityRole?: AccessibilityRole;
}
```

### MdiProps

Props for the Mdi component.

```tsx copy
import { type MdiProps } from 'rn-iconify';

interface MdiProps extends IconProps {
  name: MdiIconName;
}
```

### HeroiconsProps

```tsx copy
import { type HeroiconsProps } from 'rn-iconify';

interface HeroiconsProps extends IconProps {
  name: HeroiconsIconName;
}
```

Each icon component has its own Props type following the same pattern.

## Theme Types

### IconTheme

Theme configuration object.

```tsx copy
import { type IconTheme } from 'rn-iconify';

const theme: IconTheme = {
  size: 24,
  color: '#333333',
  placeholder: 'shimmer',
};
```

```tsx
interface IconTheme {
  /** Default icon size */
  size?: number;

  /** Default icon color */
  color?: string;

  /** Default placeholder */
  placeholder?: PlaceholderType;
}
```

### IconThemeProviderProps

```tsx copy
import { type IconThemeProviderProps } from 'rn-iconify';

interface IconThemeProviderProps {
  theme: IconTheme;
  children: ReactNode;
}
```

## Animation Types

### AnimationType

Available animation presets.

```tsx copy
import { type AnimationType } from 'rn-iconify';

type AnimationType =
  | 'spin'    // Continuous rotation
  | 'pulse'   // Scale up/down
  | 'bounce'  // Vertical bounce
  | 'shake'   // Horizontal shake
  | 'ping'    // Expand and fade
  | 'wiggle'; // Rotate wiggle
```

## Data Types

### IconData

Icon SVG data structure.

```tsx copy
import { type IconData } from 'rn-iconify';

interface IconData {
  /** SVG body content */
  body: string;

  /** Icon width */
  width: number;

  /** Icon height */
  height: number;

  /** Left offset */
  left?: number;

  /** Top offset */
  top?: number;
}
```

### IconBundle

Offline bundle structure.

```tsx copy
import { type IconBundle } from 'rn-iconify';

interface IconBundle {
  /** Bundle format version */
  version: number;

  /** Generation timestamp */
  generated?: string;

  /** Icon data keyed by full name */
  icons: Record<string, IconData>;
}
```

## Utility Types

### PrefetchOptions

Options for prefetchIcons.

```tsx copy
import { type PrefetchOptions } from 'rn-iconify';

interface PrefetchOptions {
  /** Timeout per icon (ms) */
  timeout?: number;

  /** Continue on error */
  continueOnError?: boolean;

  /** Progress callback */
  onProgress?: (loaded: number, total: number) => void;
}
```

### PrefetchResult

Result from prefetchIcons.

```tsx copy
import { type PrefetchResult } from 'rn-iconify';

interface PrefetchResult {
  /** Successfully loaded icons */
  success: string[];

  /** Failed icons with errors */
  failed: Array<{ icon: string; error: Error }>;

  /** Total duration (ms) */
  duration: number;
}
```

## Type Helpers

### InferIconName

Infer icon name type from a component.

```tsx copy
import { type InferIconName, Mdi, Heroicons } from 'rn-iconify';

type MdiName = InferIconName<typeof Mdi>;
// Equivalent to MdiIconName

type HeroName = InferIconName<typeof Heroicons>;
// Equivalent to HeroiconsIconName
```

### IconSetPrefix

Get prefix for an icon set.

```tsx copy
import { type IconSetPrefix } from 'rn-iconify';

type Prefix = IconSetPrefix<'Mdi'>; // 'mdi'
type Prefix2 = IconSetPrefix<'Heroicons'>; // 'heroicons'
```
